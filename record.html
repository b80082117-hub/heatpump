currentBase64Photos : [] }; // 4. 安全存入本地瀏覽器快取 (避免現場網路斷線導致紀錄消失) let records = JSON.parse(localStorage.getItem("hp_dual_records_v3")) || []; records.unshift(record); localStorage.setItem("hp_dual_records_v3", JSON.stringify(records)); // 5. 【關鍵上傳邏輯】非同步傳送到雲端 Google 試算表 // ⚠️ 請務必將下方的網址，替換成您在 GAS 「部署為網頁應用程式」後生成的實際部署網址 (以 /exec 結尾) const gasUrl = "https://script.google.com/macros/s/AKfycbycjmQjr0otxP-kJDnIMc2_9ZaSTs7YxQGT37HiO32u-Px6c2C0LHBgCA3w4gpx9w4Llg/exec"; // 使用 URLSearchParams 標準封裝，才能被 GAS 的 e.parameter.jsonData 讀取 const payload = new URLSearchParams(); payload.append("jsonData", JSON.stringify(record)); fetch(gasUrl, { method: "POST", body: payload, headers: { "Content-Type": "application/x-www-form-urlencoded" }, mode: "no-cors" // 跨網域(CORS)防無聲阻斷設定 }) .then(() => { alert("📥 本地存檔成功，且已同步發送至雲端試算表！"); if (typeof clearForm === "function") clearForm(); // 表單欄位清空 if (typeof renderRecords === "function") renderRecords(); // 刷新歷史紀錄清單 }) .catch(err => { console.error("雲端連線失敗:", err); alert("⚠️ 本地已成功存檔！但網路連線或雲端配置有誤，未能同步至 Google 試算表。"); if (typeof clearForm === "function") clearForm(); if (typeof renderRecords === "function") renderRecords(); }); }
